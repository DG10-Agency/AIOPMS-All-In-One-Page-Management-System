      $brand = $this->get_brand_settings();

  $images);
          }

          $styled_html = $this->apply_brand_styling($content_data, $brand);

          // 6. Create WordPress page
          $page_id = wp_insert_post([
              'post_type' => 'page',
              'post_title' => $content_data['title'],
              'post_name' => sanitize_title($content_data['title']),
              'post_content' => $styled_html,
              'post_status' => 'draft', // Let user review first
              'post_author' => get_current_user_id(),
          ]);

          // 7. Generate schema markup (existing function)
          artitechcore_generate_schema_markup($page_id);

          // 8. Set featured image (first generated image)
          if (!empty($images[0]['id'])) {
              set_post_thumbnail($page_id, $images[0]['id']);
          }

          // 9. Add custom fields/metadata
          update_post_meta($page_id, '_artitechcore_ai_generated', true);
          update_post_meta($page_id, '_artitechcore_generation_date',
  current_time('mysql'));

          // 10. Progress update
          $this->update_progress("Created: " . $content_data['title']);
      }

      // 11. Build navigation menu automatically
      $this->create_main_menu($pages);

      // 12. Set homepage (if generated)
      if (in_array('home', $pages)) {
          $homepage = get_page_by_title('Home'); // or by slug
          update_option('page_on_front', $homepage->ID);
          update_option('show_on_front', 'page');
      }
  }

  function get_unsplash_image($query, $orientation = 'landscape') {
      $access_key = 'YOUR_UNSPLASH_ACCESS_KEY';
      $url = "https://api.unsplash.com/photos/random?query=" . urlencode($query)
   . "&orientation=$orientation&client_id=$access_key";

      $response = wp_remote_get($url);
      $data = json_decode(wp_remote_retrieve_body($response), true);

      // Download image
      $image_url = $data['urls']['regular'];
      $upload = media_sideload_image($image_url, 0, $query, 'id');

      return $upload; // attachment ID
  }

  Pros: Free, beautiful, diverse, safe
  Cons: Not brand-specific, limited control

  Strategy B: DALL-E 3 (Custom, Premium)

  function generate_dalle_image($prompt, $size = '1024x1024') {
      $api_key = get_option('artitechcore_openai_api_key');

      $body = [
          'model' => 'dall-e-3',
          'prompt' => $prompt,
          'n' => 1,
          'size' => $size,
          'quality' => 'standard',
      ];

      $response = wp_remote_post('https://api.openai.com/v1/images/generations',
   [
          'headers' => [
              'Authorization' => 'Bearer ' . $api_key,
              'Content-Type' => 'application/json'
          ],
          'body' => json_encode($body)
      ]);

      $result = json_decode(wp_remote_retrieve_body($response), true);
      $image_url = $result['data'][0]['url'];

      // Download to media library
      return media_sideload_image($image_url, 0, 'AI Generated', 'id');
  }
